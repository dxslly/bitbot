function OnGameTick (ev)
    moveChance = math.random();
    
    if(moveChance < 0.1) then
		return PlantBomb;
	elseif (moveChance < 0.3) then
		return MoveUp;
    elseif (moveChance < 0.6) then
		return MoveRight;
    elseif (moveChance < 0.8) then
		return MoveDown;
    else
		return MoveLeft;
    end
end
